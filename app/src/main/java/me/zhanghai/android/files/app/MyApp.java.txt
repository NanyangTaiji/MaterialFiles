        if (mMainExecutor == null) {

    // -------------------------------------------------------------------------
    // Notification template  (BackgroundActivityStarter.kt)

    private NotificationTemplate mBackgroundActivityStartNotificationTemplate;

    public NotificationTemplate getBackgroundActivityStartNotificationTemplate() {
        if (mBackgroundActivityStartNotificationTemplate == null) {
            NotificationChannelTemplate channelTemplate = new NotificationChannelTemplate(
                    "background_activity_start",
                    R.string.notification_channel_background_activity_start_name,
                    NotificationManagerCompat.IMPORTANCE_HIGH,
                    R.string.notification_channel_background_activity_start_description,
                    /* showBadge= */ false
            );
            mBackgroundActivityStartNotificationTemplate = new NotificationTemplate(
                    channelTemplate,
                    R.color.color_primary,
                    R.drawable.notification_icon,
                    /* ongoing= */    true,
                    /* autoCancel= */ true,
                    NotificationCompat.CATEGORY_ERROR,
                    NotificationCompat.PRIORITY_HIGH
            );
        }
        return mBackgroundActivityStartNotificationTemplate;
    }

    /**
     * Ordered list of initialisation steps, mirroring {@code appInitializers} in
     * AppInitializers.kt. AppUpgrader is intentionally omitted.
     */
    private void runInitializers() {
        initializeCrashlytics();
        disableHiddenApiChecks();
        initializeWebViewDebugging();
        initializeCoil();
        initializeFileSystemProviders();
        // upgradeApp() — excluded per request
        initializeLiveDataObjects();
        initializeCustomTheme();
        initializeNightMode();
        createNotificationChannels();
    }

    private void initializeCrashlytics() {
        // #ifdef NONFREE
        // me.zhanghai.android.files.nonfree.CrashlyticsInitializer.initialize();
        // #endif
    }

    private void disableHiddenApiChecks() {
        HiddenApi.disableHiddenApiChecks();
    }

    private void initializeWebViewDebugging() {
        if (BuildConfig.DEBUG) {
            WebView.setWebContentsDebuggingEnabled(true);
        }
    }

    @SuppressWarnings("deprecation") // AsyncTask.THREAD_POOL_EXECUTOR still fine for fire-and-forget
    private void initializeFileSystemProviders() {
        FileSystemProviders.install();
        FileSystemProviders.setOverflowWatchEvents(true);

        // SingletonContext.init() calls NameServiceClientImpl.initCache() which hits the network,
        // so push it off the main thread exactly as the original Kotlin does.
        AsyncTask.THREAD_POOL_EXECUTOR.execute(() -> {
            Properties props = new Properties();
            props.setProperty("jcifs.netbios.cachePolicy", "0");
            props.setProperty("jcifs.smb.client.maxVersion", "SMB1");
            try {
                SingletonContext.init(props);
            } catch (Exception e) {
                e.printStackTrace();
            }
        });

        FtpClient.setAuthenticator(FtpServerAuthenticator.INSTANCE);
        SftpClient.setAuthenticator(SftpServerAuthenticator.INSTANCE);
        SmbClient.setAuthenticator(SmbServerAuthenticator.INSTANCE);
        WebDavClient.setAuthenticator(WebDavServerAuthenticator.INSTANCE);
    }

    private void initializeCoil() {
        CoilKt.initializeCoil(this);
    }

    private void initializeLiveDataObjects() {
        // Force init on the main thread so it never happens on a background thread.
        StorageVolumeListLiveData.INSTANCE.getValue();
        Settings.FILE_LIST_DEFAULT_DIRECTORY.getValue();
    }

    private void initializeCustomTheme() {
        CustomThemeHelper.initialize(this);
    }

    private void initializeNightMode() {
        NightModeHelper.initialize(this);
    }

    private void createNotificationChannels() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            List<android.app.NotificationChannel> channels = Arrays.asList(
                    getBackgroundActivityStartNotificationTemplate()
                            .getChannelTemplate().create(this),
                    FileJobKt.getFileJobNotificationTemplate()
                            .getChannelTemplate().create(this),
                    FtpServerKt.getFtpServerServiceNotificationTemplate()
                            .getChannelTemplate().create(this)
            );
            getNotificationManagerCompat().createNotificationChannels(channels);
        }
    }

    // -------------------------------------------------------------------------
    // BackgroundActivityStarter  (BackgroundActivityStarter.kt)
    // -------------------------------------------------------------------------

    /**
     * Start {@code intent} immediately if the app is in the foreground, otherwise
     * post a high-priority notification that lets the user tap to launch it.
     */
    public void startBackgroundActivity(
            Intent intent, CharSequence title, CharSequence text, Context context) {
        intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        if (isAppInForeground()) {
            IntentUtils.startActivitySafe(context, intent);
        } else {
            notifyStartActivity(intent, title, text, context);
        }
    }

    private boolean isAppInForeground() {
        return ProcessLifecycleOwner.get()
                .getLifecycle()
                .getCurrentState()
                .isAtLeast(Lifecycle.State.STARTED);
    }

    // We intentionally don't guard with a permission check here — the original
    // Kotlin code accepts a potential SecurityException from the file-job layer.
    @SuppressLint("MissingPermission")
    private void notifyStartActivity(
            Intent intent, CharSequence title, CharSequence text, Context context) {
        int flags = PendingIntent.FLAG_ONE_SHOT | PendingIntent.FLAG_CANCEL_CURRENT;
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
            flags |= PendingIntent.FLAG_IMMUTABLE;
        }
        PendingIntent pendingIntent = PendingIntent.getActivity(
                context, intent.hashCode(), intent, flags);

        android.app.Notification notification =
                getBackgroundActivityStartNotificationTemplate()
                        .createBuilder(context)
                        .setContentTitle(title)
                        .setContentText(text)
                        .setContentIntent(pendingIntent)
                        .build();

        getNotificationManagerCompat().notify(intent.hashCode(), notification);
    }
