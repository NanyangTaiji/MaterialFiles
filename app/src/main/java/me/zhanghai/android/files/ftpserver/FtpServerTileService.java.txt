package me.zhanghai.android.files.ftpserver;

import android.os.Build;
import android.service.quicksettings.Tile;
import android.service.quicksettings.TileService;

import androidx.annotation.RequiresApi;
import androidx.lifecycle.Observer;

import me.zhanghai.android.files.compat.TileServiceCompatKt;

@RequiresApi(Build.VERSION_CODES.N)
public class FtpServerTileService extends TileService {
    private final Observer<FtpServerService.State> observer = this::onFtpServerStateChanged;

    @Override
    public void onStartListening() {
        super.onStartListening();
        FtpServerService.getStateLiveData().observeForever(observer);
    }

    @Override
    public void onStopListening() {
        super.onStopListening();
        FtpServerService.getStateLiveData().removeObserver(observer);
    }

    private void onFtpServerStateChanged(FtpServerService.State state) {
        Tile tile = getQsTile();
        switch (state) {
            case STARTING:
            case RUNNING:
                tile.setState(Tile.STATE_ACTIVE);
                break;
            case STOPPING:
                tile.setState(Tile.STATE_UNAVAILABLE);
                break;
            case STOPPED:
                tile.setState(Tile.STATE_INACTIVE);
                break;
        }
        tile.updateTile();
    }

    @Override
    public void onClick() {
        super.onClick();
        if (isLocked()) {
            unlockAndRun(this::toggle);
        } else {
            toggle();
        }
    }

    private void toggle() {
        TileServiceCompatKt.doWithStartForegroundServiceAllowed(
                this, () -> FtpServerService.toggle(this));
    }
}
