package me.zhanghai.android.files.storage;

import android.app.Activity;
import android.content.Context;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;

import androidx.activity.result.contract.ActivityResultContract;
import androidx.fragment.app.FragmentActivity;
import androidx.fragment.app.FragmentTransaction;

import me.zhanghai.android.files.app.AppActivity;
import me.zhanghai.android.files.util.ArgsUtil;
import me.zhanghai.android.files.util.IntentUtil;
import me.zhanghai.android.files.util.IntentUtils;

public class EditSmbServerActivity extends AppActivity {

    private EditSmbServerFragment.Args args;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(findViewById(android.R.id.content)); // Assuming android.R.id.content is equivalent to android.R.layout.content in Java

        args = ArgsUtil.getArgs(this, EditSmbServerFragment.Args.class);

        if (savedInstanceState == null) {
            EditSmbServerFragment fragment = new EditSmbServerFragment();
            Bundle bundle = new Bundle();
            bundle.putSerializable("args", args);
            fragment.setArguments(bundle);

            getSupportFragmentManager().beginTransaction()
                    .add(android.R.id.content, fragment)
                    .commit();
        }
    }

    public static class EditSmbServerContract extends ActivityResultContract<EditSmbServerFragment.Args, Boolean> {

        @Override
        public Intent createIntent(Context context, EditSmbServerFragment.Args input) {
            return IntentUtils.createIntent(context, EditSmbServerActivity.class)
                    .putExtra("args", input);
        }

        @Override
        public Boolean parseResult(int resultCode, Intent intent) {
            return resultCode == Activity.RESULT_OK;
        }
    }
}
