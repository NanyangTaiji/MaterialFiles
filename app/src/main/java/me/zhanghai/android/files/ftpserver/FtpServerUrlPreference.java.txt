package me.zhanghai.android.files.ftpserver;

import android.content.Context;
import android.util.AttributeSet;
import android.view.ContextMenu;
import android.view.Menu;
import android.view.View;

import androidx.annotation.AttrRes;
import androidx.annotation.StyleRes;
import androidx.lifecycle.Observer;
import androidx.preference.Preference;
import androidx.preference.PreferenceViewHolder;

import me.zhanghai.android.files.R;
import me.zhanghai.android.files.app.ApplicationKt;
import me.zhanghai.android.files.settings.Settings;
import me.zhanghai.android.files.util.ClipboardUtilsKt;
import me.zhanghai.android.files.util.LiveDataUtilsKt;
import me.zhanghai.android.files.util.RuntimeBroadcastReceiver;

public class FtpServerUrlPreference extends Preference {
    private final Observer<Object> observer = o -> updateUrl();
    private final RuntimeBroadcastReceiver receiver;

    private String url;

    public FtpServerUrlPreference(Context context) {
        super(context);
        receiver = FtpServerUrl.createChangeReceiver(context, this::updateUrl);
        init();
    }

    public FtpServerUrlPreference(Context context, AttributeSet attrs) {
        super(context, attrs);
        receiver = FtpServerUrl.createChangeReceiver(context, this::updateUrl);
        init();
    }

    public FtpServerUrlPreference(Context context, AttributeSet attrs,
            @AttrRes int defStyleAttr) {
        super(context, attrs, defStyleAttr);
        receiver = FtpServerUrl.createChangeReceiver(context, this::updateUrl);
        init();
    }

    public FtpServerUrlPreference(Context context, AttributeSet attrs,
            @AttrRes int defStyleAttr, @StyleRes int defStyleRes) {
        super(context, attrs, defStyleAttr, defStyleRes);
        receiver = FtpServerUrl.createChangeReceiver(context, this::updateUrl);
        init();
    }

    private void init() {
        setPersistent(false);
        updateUrl();
    }

    @Override
    public void onAttached() {
        super.onAttached();
        Settings.FTP_SERVER_ANONYMOUS_LOGIN.observeForever(observer);
        Settings.FTP_SERVER_USERNAME.observeForever(observer);
        Settings.FTP_SERVER_PORT.observeForever(observer);
        receiver.register();
    }

    @Override
    public void onDetached() {
        super.onDetached();
        Settings.FTP_SERVER_ANONYMOUS_LOGIN.removeObserver(observer);
        Settings.FTP_SERVER_USERNAME.removeObserver(observer);
        Settings.FTP_SERVER_PORT.removeObserver(observer);
        receiver.unregister();
    }

    private void updateUrl() {
        url = FtpServerUrl.getUrl();
        setSummary(url != null ? url
                : getContext().getString(R.string.ftp_server_url_summary_no_local_inet_address));
    }

    @Override
    public void onBindViewHolder(PreferenceViewHolder holder) {
        super.onBindViewHolder(holder);
        holder.itemView.setOnCreateContextMenuListener((menu, view, menuInfo) -> {
            if (url == null) {
                return;
            }
            menu.setHeaderTitle(url);
            final String capturedUrl = url;
            menu.add(Menu.NONE, Menu.NONE, Menu.NONE, R.string.ftp_server_url_menu_copy_url)
                    .setOnMenuItemClickListener(item -> {
                        ClipboardUtilsKt.copyText(
                                ApplicationKt.getClipboardManager(), capturedUrl, getContext());
                        return true;
                    });
            if (!LiveDataUtilsKt.getValueCompat(Settings.FTP_SERVER_ANONYMOUS_LOGIN)) {
                String password = LiveDataUtilsKt.getValueCompat(Settings.FTP_SERVER_PASSWORD);
                if (!password.isEmpty()) {
                    final String capturedPassword = password;
                    menu.add(Menu.NONE, Menu.NONE, Menu.NONE,
                            R.string.ftp_server_url_menu_copy_password)
                            .setOnMenuItemClickListener(item -> {
                                ClipboardUtilsKt.copyText(
                                        ApplicationKt.getClipboardManager(),
                                        capturedPassword, getContext());
                                return true;
                            });
                }
            }
        });
    }
}
